## application\(graph\)

Connecting Graph问题的总结

• 并查集原生操作: 

•查询两个元素是否在同一个集合内   
•合并两个元素所在的集合

• 并查集的派生操作: 

•查询某个元素所在集合的元素个数   
•查询当前集合的个数

            root_i = find(i)
            root_j = find(j)

                self.count -= 1

            connect(i,j)

class Solution:
        
    def dfs(self, grid, i, j):
        if i<0 or j<0 or i>=len(grid) or j>=len(grid[0]) or grid[i][j] != '1':
            return
        grid[i][j] = '#'
        self.dfs(grid, i+1, j)
        self.dfs(grid, i, j+1)
        self.dfs(grid, i-1, j)
        self.dfs(grid, i, j-1)
        
    def numIslands(self, grid: List[List[str]]) -> int:
        row = len(grid)
        col = len(grid[0])
        count = 0
        for i in range(row):
            for j in range(col):
                if grid[i][j] == '1':
                    self.dfs(grid, i,j)
                    count += 1
        return count

        root_b = self.find(b)
        if root_a != root_b:
            self.count -= 1
            self.father[root_a] = root_b
            
class Solution:
    def numIslands2(self, m: int, n: int, positions: List[List[int]]) -> List[int]:
        def is_island(x,y):
            if x<0 or x>=m or y<0 or y>= n or uf.father[x*n+y] == -1:
                return False
            return True
        
        uf = UnionFind(m,n)
        res = []
        for new in positions:
            if uf.father[new[0]*n+new[1]] == -1:
                uf.father[new[0]*n+new[1]] = new[0]*n+new[1]
                uf.count += 1
            for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:

### Number of Distinct Islands

```python
'''
dfs
shapes = set(unfrozenset((0,0),(0,-1))...)
return len(shapes)
time :O(mn)
space: O(mn)
'''
class Solution:
    def numDistinctIslands(self, grid: List[List[int]]) -> int:
        shapes = set()
        m,n = len(grid), len(grid[0])
        
        def dfs(i,j, i0, j0, shape):
            if i < 0 or j <0 or i>=m or j >= n or grid[i][j] != 1:
                return
            grid[i][j] = 0
            shape.add((i-i0,j-j0))
            dfs(i+1, j, i0, j0, shape)
            dfs(i-1, j, i0, j0, shape)
            dfs(i, j+1, i0, j0, shape)
            dfs(i, j-1, i0, j0, shape)
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    shape = set()
                    dfs(i,j, i, j, shape)
                    # print(shape)
                    if shape:
                        shapes.add(frozenset(shape))
        return len(shapes)
```
